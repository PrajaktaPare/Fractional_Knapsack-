C Code 


#include 

struct Item {
    int value;
    int weight;
    float ratio;
};

int compare(const void *a, const void *b) {
    struct Item *item1 = (struct Item*)a;
    struct Item *item2 = (struct Item*)b;
    return (item2->ratio > item1->ratio) - (item1->ratio > item2->ratio);
}

void fractionalKnapsack(int W, struct Item arr[], int n) {
    qsort(arr, n, sizeof(arr[0]), compare);

    float totalValue = 0.0;

    for (int i = 0; i < n; i++) {
        if (arr[i].weight <= W) {
            W -= arr[i].weight;
            totalValue += arr[i].value;
        } else {
            totalValue += arr[i].value * ((float)W / arr[i].weight);
            break;
        }
    }
    printf("Maximum Value in Knapsack = %.2f\n", totalValue);
}

int main() {
    struct Item arr[] = { {60, 10, 6}, {100, 20, 5}, {120, 30, 4} };
    int W = 50;
    int n = sizeof(arr) / sizeof(arr[0]);
    
    for (int i = 0; i < n; i++) {
        arr[i].ratio = (float) arr[i].value / arr[i].weight;
    }
    fractionalKnapsack(W, arr, n);
    return 0;
}

    }

function fractionalKnapsack(W, items) {
    items.sort((a, b) => b.ratio - a.ratio);

    let totalValue = 0;

    for (let i = 0; i < items.length; i++) {
        if (items[i].weight <= W) {
            W -= items[i].weight;
            totalValue += items[i].value;
        } else {
            totalValue += items[i].value * (W / items[i].weight);
            break;
        }
    }
    console.log('Maximum Value in Knapsack = ' + totalValue.toFixed(2));
}
